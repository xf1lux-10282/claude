# PX1V-PtPd-v20 インストールコマンド

# 1. v20をインストール
sudo cp /Users/daisukekinoshita/Desktop/PX1V-PtPd-v20.quad /Library/Printers/QTR/quadtone/QuadP700/

# 2. QTRカーブデータベースを更新
/Library/Printers/QTR/bin/quadcurves QuadP700

# 3. QTR Print-Toolの設定をリセット（オプション）
defaults delete com.quadtonerip.QTR-Print-Tool

# これで準備完了！
# QTR Print-Toolで "PX1V-PtPd-v20" を選択してテストプリント
